### 2. src/renderer/components/SurferWebView.tsx (–Ω–æ–≤—ã–π —Ñ–∞–π–ª)

```typescript
import React, { useEffect, useRef } from 'react';
import { Browser } from '@zen-browser/surfer';

interface SurferWebViewProps {
  url: string;
  isActive: boolean;
  onNavigate?: (url: string) => void;
  onTitleChange?: (title: string) => void;
  onFaviconChange?: (favicon: string) => void;
  onLoadStart?: () => void;
  onLoadEnd?: () => void;
}

export const SurferWebView: React.FC<SurferWebViewProps> = ({
  url,
  isActive,
  onNavigate,
  onTitleChange,
  onFaviconChange,
  onLoadStart,
  onLoadEnd
}) => {
  const containerRef = useRef<HTMLDivElement>(null);
  const browserRef = useRef<Browser | null>(null);

  useEffect(() => {
    if (!containerRef.current) return;

    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è Surfer browser
    const browser = new Browser({
      container: containerRef.current,
      url: url
    });

    // –°–æ–±—ã—Ç–∏—è –Ω–∞–≤–∏–≥–∞—Ü–∏–∏
    browser.on('navigate', (newUrl: string) => {
      onNavigate?.(newUrl);
    });

    browser.on('title-change', (title: string) => {
      onTitleChange?.(title);
    });

    browser.on('favicon-change', (favicon: string) => {
      onFaviconChange?.(favicon);
    });

    browser.on('load-start', () => {
      onLoadStart?.();
    });

    browser.on('load-end', () => {
      onLoadEnd?.();
    });

    browserRef.current = browser;

    return () => {
      browser.destroy();
    };
  }, []);

  // –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ URL –ø—Ä–∏ –∏–∑–º–µ–Ω–µ–Ω–∏–∏
  useEffect(() => {
    if (browserRef.current && url) {
      browserRef.current.loadURL(url);
    }
  }, [url]);

  return (
    <div
      ref={containerRef}
      style={{
        width: '100%',
        height: '100%',
        display: isActive ? 'block' : 'none'
      }}
    />
  );
};
```
